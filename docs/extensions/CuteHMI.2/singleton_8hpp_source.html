105 template <class C>
106 C & Singleton<C>::Instance()
107 {
108  //<CuteHMI-2.workaround target="std" cause="design">
109  // Function std::unique_ptr::reset() sets internal pointer to nullptr and only after that it will delete its contents. This
110  // causes error, when managed object still needs to be accessed through std::unique_ptr::get() function by members of managed
111  // object during their destruction. Workaround is to use additional raw pointer. In general this prevents std::unique_ptr from
112  // being used in inconsistent state (calling std::unique_ptr::get() while inside std::unique_ptr::reset()).
113  static C * instancePtr = InstancePtr().get();
114  return *instancePtr;
115  //</CuteHMI-2.workaround>
116 }

118 template <class C>
119 void Singleton<C>::Destroy()
120 {
121  //<CuteHMI-2.workaround>
122  // A bit paranoic, but if Instance() wasn't called before, it would point to nullptr. What if it is used by a dtor of one of the
123  // members of managed object? After InstancePtr().reset(), std::unique_ptr::reset() first sets internal pointer to nullptr, then
124  // calls destructor (hence Instance() would point to nullptr during destruction).
125  Instance();
126  //</CuteHMI-2.workaround>
127  InstancePtr().reset();
128 }
